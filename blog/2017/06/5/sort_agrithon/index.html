						indicator = 1; break;
					}
				}

						{
							k++;
						}

						{
							temp[k] = temp[k - 1];
						}
						temp[k] = vec[i];
					}

						temp[j + 1] = vec[i];
					}
				}
				else

					temp[i * 2] = vec[i];
